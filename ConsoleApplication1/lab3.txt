int main() {

	std::cout << "Starting" << std::endl;

	GLFWwindow* window;

	// Initialize GLFW
	if (!glfwInit()) {
		std::cout << "GLFW could not start!" << std::endl;
		return -1;
	}

	// Set OpenGL versions
	glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
	glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
	glfwWindowHint(GLFW_OPENGL_PROFILE,
		GLFW_OPENGL_CORE_PROFILE);

	// Create a window
	//window = glfwCreateWindow(800, 400, "Hello Window", NULL, NULL);
	//window = glfwCreateWindow(1024, 768, "Pathum Madhusanka", NULL, NULL);
	window = glfwCreateWindow(800, 600, "Lab 03", NULL, NULL);


	if (window == NULL) {
		std::cout << "Window could not be created!" << std::endl;
		glfwTerminate();
		return -1;
	}

	// Make window context current
	glfwMakeContextCurrent(window);

	// Initialize GLAD
	if (!gladLoadGLLoader((GLADloadproc)glfwGetProcAddress)) {

		glfwTerminate();
		return -1;
	}

	//Vertex Shader
	unsigned int vertexShader;
	vertexShader = glCreateShader(GL_VERTEX_SHADER);
	glShaderSource(vertexShader, 1,
		&vertexShaderSource, NULL);
	glCompileShader(vertexShader);

	//Fragment Shader
	unsigned int fragmentShader;
	fragmentShader = glCreateShader(GL_FRAGMENT_SHADER);
	glShaderSource(fragmentShader, 1,
		&fragmentShaderSource, NULL);
	glCompileShader(fragmentShader);

	// Shader Program
	unsigned int shaderProgram;

	shaderProgram = glCreateProgram();

	glAttachShader(shaderProgram, vertexShader);
	glAttachShader(shaderProgram, fragmentShader);

	glLinkProgram(shaderProgram);

	glDeleteShader(vertexShader);
	glDeleteShader(fragmentShader);

	// Vertex Data 
	float vertices[] =
	{
		-0.5f,0.5f,0.0f
	};


	unsigned int VBO, VAO;

	glGenVertexArrays(1, &VAO);
	glGenBuffers(1, &VBO);

	glBindVertexArray(VAO);

	glBindBuffer(GL_ARRAY_BUFFER, VBO);

	glBufferData(GL_ARRAY_BUFFER,
		sizeof(vertices),
		vertices,
		GL_STATIC_DRAW);

	glVertexAttribPointer(
		0,
		3,
		GL_FLOAT,
		GL_FALSE,
		3 * sizeof(float),
		(void*)0);

	glEnableVertexAttribArray(0);

	// NOTE!!!! Enable point size in the vertex shader
	glEnable(GL_PROGRAM_POINT_SIZE);


	// Set the viewport
	//glViewport(0, 0, 800, 200);
	//glViewport(0, 0, 1024, 768);


	// Render loop
	while (!glfwWindowShouldClose(window)) {

		//processInput(window);

		// Clear the window
		//glClearColor(1.0f, 0.0f, 0.0f, 1.0f);
		//glClearColor(0.0f, 0.0f, 1.0f, 1.0f);
		//glClearColor(0.2f, 0.3f, 0.4f, 1.0f);
		//glClearColor(1.0f, 0.0f, 0.0f, 1.0f);
		//glClearColor(0.0f, 1.0f, 0.0f, 1.0f);
		//glClearColor(0.0f, 0.0f, 1.0f, 1.0f);
		//glClearColor(red, green, blue, 1.0f);
		glClearColor(0.2f, 0.3f, 0.4f, 1.0f);

		glClear(GL_COLOR_BUFFER_BIT);

		glUseProgram(shaderProgram);

		glBindVertexArray(VAO);

		glDrawArrays(GL_POINTS, 0, 1);

		// Swap buffers
		glfwSwapBuffers(window);

		// Check events(keyboard, mouse)
		glfwPollEvents();
	}

	// Clean up and exit
	glfwTerminate();
	return 0; 
}
